Write a while statement to count from 1 to 100 by 3s

/*this program is counting numbers
from 1 to 100 by 3s*/
#include<stdio.h>

int main(void)
{
  int k=1;

  while(k <= 100)
{
     printf("%d\n", k);
     k+=3; //making an interval of 3 up to 100
}

  return 0;

}
